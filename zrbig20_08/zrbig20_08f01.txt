FORM set_init_value .

  CLEAR : s_carrid.
  s_carrid-sign   = 'I'.
  s_carrid-option = 'EQ'.
  s_carrid-low    = 'LH'.
  APPEND s_carrid.


ENDFORM.

  ON  b~connid EQ c~connid
  INNER JOIN sbook AS d

FORM manipulate_display .

  LOOP AT SCREEN.
    IF p_carr EQ 'X'.
      IF screen-group1 EQ 'CAR'.
        screen-active = 1.
      ENDIF.
    ELSE.
      IF screen-group1 EQ 'CAR'.
        screen-active = 0.
      ENDIF.
    ENDIF.

    IF p_conn EQ 'X'.
      IF screen-group1 EQ 'CON'.
        screen-active = 1.
      ENDIF.
    ELSE.
      IF screen-group1 EQ 'CON'.
        screen-active = 0.
      ENDIF.
    ENDIF.

    MODIFY SCREEN.
  ENDLOOP.

ENDFORM.
